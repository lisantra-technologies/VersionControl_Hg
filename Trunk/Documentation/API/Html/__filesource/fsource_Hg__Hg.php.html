  37. /**
  38.  * Base class to begin the fluent API
  39.  *
  40.  * This package interfaces with the Mercurial command-line binary, which
  41.  * must be installed on the same system as this package. The author of Mercurial
  42.  * is on record preferring that all non-python programs interface with the
  43.  * CLI binary.
  44.  *
  45.  * There are no C-bindings, so a PECL extension is unlikely, as is a pure PHP
  46.  * implementation due to the tremendous workload involved in keeping up with
  47.  * changes Mercurial.
  48.  *
  49.  * Usage:
  50.  * <code>
  51.  * require_once 'VersionControl/Hg.php';
  52.  * $hg = new VersionControl_Hg('/path/to/repository');
  53.  * </code>
  54.  *
  55.  * Setting the repository also automatically finds and sets the local
  56.  * path of the Mercurial binary it will use. If multiple installations of
  57.  * Mercurial are found, this package will employ only the first found on the
  58.  * system path.
  59.  *
  60.  * You may also provide a location of a repository after instantiation:
  61.  * <code>
  62.  * require_once 'VersionControl/Hg.php';
  63.  * $hg = new VersionControl_Hg();
  64.  * $hg->setRepository('/path/to/repository');
  65.  * </code>
  66.  * or
  67.  * <code>$hg->repository = '/path/to/repository';</code>
  68.  *
  69.  * Calling all commands other than 'version' without having already set a

  71.  *
  72.  * PHP version 5
  73.  *
  74.  * @category  VersionControl

  84.     /**
  85.      * Constructor
  86.      *
  87.      * Assumes it will be working on a local filesystem repository
  88.      *
  89.      * The repository path is optional in the constructor.
  90.      * It can be set with an explicit setting of the repository's path.
  91.      * You can only set the path once since Repository is a singleton;
  92.      * All subsqeuent attempts will silently fail after one of the below
  93.      * has been performed:
  94.      * <code>$hg->setRepository('/path/to/repository');</code>
  95.      * or
  96.      * <code>$hg->repository->setPath('/path/to/repository');</code>
  97.      * or
  98.      * <code>$hg->repository = '/path/to/repository';</code>
  99.      *
  100.      * @param string $repository Optional path to a mercurial repository.
  101.      *
  102.      * @return void 
  103.      */
  104.     public function __construct($repository = null)
  105.     {
  106.         /* invalid repository will trigger an exception in the
  107.          * child class VersionControl_Hg_Container_Repository */

  109.  
  110.         /* This is second, since why look for an executable if the passed

  119.     /**
  120.      * Proxy down to the command class
  121.      *
  122.      * This also allows programmers to use both
  123.      * <code>$executables_object = $hg->executable;</code>
  124.      * and
  125.      * <code>$executables_object = $hg->getExecutable();</code>
  126.      * to both return an instance of VersionControl_Hg_Executable, for example.
  127.      *
  128.      * @param string $method    The function being called
  129.      * @param mixed  $arguments The parameters passed to that function
  130.      *
  131.      * @return VersionControl_Hg_Command_Abstract 
  132.      */
  133.     public function __call($method, $arguments)
  134.     {
  135.         $value = ( empty($arguments) ) ? null : $arguments[0];
  136.  
  137.         $possible_prefix = strtolower(substr($method, 0, 3));
  138.         $possible_object = strtolower(substr($method, 3));
  139.  
  140.         /* Very limited use:
  141.          * $hg->getExecutable()->getPath() = $hg->executable->getPath()
  142.          */
  143.         switch ( $possible_prefix ) {
  144.             case 'set':
  145.                 if ( $possible_object === 'repository' ) {
  146.                     return VersionControl_Hg_Container_Repository::getInstance(
  147.                         $this, $value
  148.                     );
  149.                 } elseif ( $possible_object === 'executable' ) {
  150.                     return VersionControl_Hg_Executable::getInstance($this, $value);
  151.                 } else {
  152.                     throw new ErrorException(
  153.                         "set{$possible_object} is not implemented"
  154.                     );
  155.                 }
  156.                 break;
  157.             case 'get':
  158.                 if ( $possible_object === 'repository' ) {
  159.                     return VersionControl_Hg_Container_Repository::getInstance();
  160.                 } elseif ( $possible_object === 'executable' ) {
  161.                     return VersionControl_Hg_Executable::getInstance();
  162.                 } else {
  163.                     throw new ErrorException(
  164.                         "get{$possible_object} is not implemented"
  165.                     );
  166.                 }
  167.                 break;
  168.             /* proxy to Hg/Command.php */
  169.             default:
  170.                 /* must pass an instance of VersionControl_Hg to provide it with
  171.                  * the executable and repository */
  172.                 $command = new VersionControl_Hg_CommandProxy($this);
  173.                 return call_user_func_array(array($command, $method), $arguments);
  174.                 break;
  175.         }
  176.     }

  178.     /**
  179.      * Returns an object, usually handled by a command
  180.      *
  181.      * A $name is a lowercase, short name of the object:
  182.      * $hg->executable is an instance of VersionControl_Hg_Executable and can
  183.      * be echoed to invoke __toString() to get a pertinent piece of metadata.
  184.      *
  185.      * Instead of calling:
  186.      * <code>$hg->getVersion();</code>,
  187.      * we simplify:
  188.      * <code>$version = $hg->version</code>.

  191.      *
  192.      * @return mixed 
  193.      */
  194.     public function __get($name)
  195.     {
  196.         /* Instantiate the object corresponding to the short name
  197.          * most are commands, some are top-level objects */
  198.         switch ($name) {
  199.             case 'repository':
  200.                 /* Singleton let's us use an instance or create a new one if
  201.                  * not instantiated
  202.                  *
  203.                  * We're ok with a null argument here since to even use
  204.                  * this, $hg would already have to be instantitated
  205.                  * successfully with a repo argument.
  206.                  */
  207.                 return VersionControl_Hg_Container_Repository::getInstance();
  208.                 break;
  209.             case 'executable':
  210.                 /* Singleton let's us use an instance or create a new one if
  211.                  * not instantiated */
  212.                 return VersionControl_Hg_Executable::getInstance();//null, $this
  213.                 break;
  214.             default:
  215.                 /* its a command */
  216.                 $command = new VersionControl_Hg_CommandProxy($this);
  217.                 return call_user_func_array(array($command, $name), array());
  218.                 break;
  219.         }
  220.     }

  247.             default:
  248.                 /* its a command */
