  237.     {
  238.         /* is it a pattern or a simple array of files?
  239.          * the scheme must be the very first key.

  301.     public function to($directory = '.')
  302.     {
  303.         /* test path's validity */
  304.         if ( $directory != realpath($directory) ) {

  306.                 VersionControl_Hg_Command_Exception::BAD_ARGUMENT,
  307.                 "The path '{$directory}' does not seem to exist on

  310.         }
  311.  
  312.         $output_template = $this->getOption('output');
  313.  
  314.         /* append the directory to --output */
  315.         $output_template = $directory . DIRECTORY_SEPARATOR . $output_template;
  316.  
  317.         /* now, add it back to the stack of options... */
